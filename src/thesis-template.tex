% Head and foot lines
\usepackage{fancyhdr}

\pagestyle{fancy}
\fancyhf{} % clear all header and foot
\fancyhead[LE,RO]{\leftmark} % left header: chaptername
\fancyfoot[LE]{\hspace{\pagenumberposition}\thepage} % pagenumber for even pages 
\fancyfoot[RO]{\thepage\hspace{\pagenumberposition}} % pagenumber for uneven pages

% Chapterpages use style plain, configure in the same way

    \fancyhf{} 
    \fancyhead[LE,RO]{\leftmark} 
    \fancyfoot[LE]{\hspace{\pagenumberposition}\thepage} 
    \fancyfoot[RO]{\thepage\hspace{\pagenumberposition}}
}
